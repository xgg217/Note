# 下拉刷新与上拉加载更多

## 示例

```vue
<template>
  <van-pull-refresh v-model="isLoading" @refresh="onRefresh">
    <van-list
      v-model="loading"
      :finished="finished"
      :finished-text="$t('noMore')"
      @load="onLoad"
      class="productBox"
      :immediate-check="false"
    >
      <product-cmp :list="list" v-if="list.length > 0" />
      <van-empty v-else />
    </van-list>
  </van-pull-refresh>
</template>

<script>
import { PullRefresh, List   } from 'vant';
export default {
  data() {
    return {
      loading: false,
      finished: false,
      isLoading: false,
      // 商品列表
      list: [],

      // 请求参数
      queryObj: {
        // Language: 2, // long 1. 英文，2.繁體，3.簡體
        ShopId: 0,// long 店鋪ID
        Page: 1, //int  页码
        PageSize:10, // int  每页显示数量
        CurrentServerTime: '' //datetime 首次加载时设置的系统时间（請傳遞比如yyyy-MM-dd HH:mm:ss）
      },
    };
  },

  created() {
    const { id, title } = this.$route.query;

    this.queryObj.ShopId = id;
    this.queryObj.Page = 1;

    this.onRefresh();
  },
  
  methods: {
    // 刷新
    onRefresh() {
      console.log("刷新");
      const obj = {
        ...this.queryObj,
        Page: 1
      };
      this.asyncGetProductListByShopIdVJ(obj).then(res => {
        this.list = res;
      }).catch(err => {
        console.error(err);
      });
    },

    // 加载更多
    onLoad() {
      console.log("到了底部");

      const list = this.list;
      this.queryObj.Page = this.queryObj.Page + 1;
      this.asyncGetProductListByShopIdVJ(this.queryObj).then(res => {
        this.list = [...list, ...res];
      }).catch(err => {
        console.error(err);
      })
    },

    // 获取指定店铺的商品列表
    asyncGetProductListByShopIdVJ(obj) {
      const query = {
        ...obj,
        CurrentServerTime: this.$moment().format("yyyy-MM-DD HH:mm")
      };
      return GetProductListByShopIdVJ(query).then(res => {
        console.log(res);
        // 查看数据是否已全部加载完毕
        if(res.length < obj.PageSize) {
          this.finished = true;
        }

        return res;
      }).catch(err => {
        console.error(err);
      }).finally(() => {
        this.isLoading = false;
      });
    }
  },
}
</script>

```
