# 基础

## 概述

+ 类型别名用来给一个类型起个新名字

 ```ts
  type 类型名 = ...
  ```

 ```ts
  type Gender = '男' | '女'
  type Obj = {
    name:string
    age:number
    gender:Gender
  }

  let newObj:Obj

  newObj = {
    name: '小刚刚',
    age: 111,
    gender: '男'
  }
  ```

## 作用

+ 类型别名常用于 **联合类型**

 ```ts
  type Name = string;
  type NameResolver = () => string;
  type NameOrResolver = Name | NameResolver;
  function getName(n: NameOrResolver): Name {
    if (typeof n === 'string') {
      return n;
    } else {
      return n();
    }
  }
  ```
