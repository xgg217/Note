# 上传

## 立即上传

- html

    ```html
    <el-upload
      :show-file-list="false"
      action="http://192.168.0.16:88/api/FormInfo/UploadByForm/"
      name="files"
      :multiple="true"
      ref="upload"
      list-type="picture-card"
      accept=".jpg,.jpeg,.png,"
      :on-success="successChange"
      :before-upload="beforeUploadFile"
      :on-error="errorChange"
    >
      <i slot="default" class="el-icon-plus"></i>
    </el-upload>
    ```

- JS

    ```js
    // 上传文件之前的钩子, 参数为上传的文件,
    beforeUploadFile() {

    },

    // 上传成功
    successChange(response, file, fileList) {
      // response 后端返回的地址
    }
    ```

## 手动控制上传

- html

    ```html
    <el-upload
      class="upload-demo"
      ref="uploads"
      name="files"
      multiple
      :file-list="fileList"
      :on-change="handleChange"
      action="http://192.168.0.16:88/api/FormInfo/UploadByForm/"
      :show-file-list="false"
      :on-success="onSuccess"
      :on-error="onError"
      :auto-upload="false"
    >
      <el-button slot="trigger" size="small" type="primary">选择附件</el-button>
      <el-button style="margin-left: 10px;" size="small" type="success" @click="handleSubmit">开始上传</el-button>
    </el-upload>
    ```

    ```js
    // 可能回同是触发多次
    onSuccess(res, file, fileList) {
      // 内部需要处理

      // res 后端返回的地址
    },

    // 选择文件--还未上传到服务器
    handleChange() {

    },

    // 上传到服务器 触发 onSuccess 函数（）
    handleSubmit() {
      this.$refs.uploads.submit();
    }
    ```
