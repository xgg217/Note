    img{

            }

    ```javascript
    const tz = document.querySelector('.tz');

            tz.addEventListener('dragover', function(e) {
                // 阻止默认事件 才会触发drop事件
                e.preventDefault();
                // e.
            }, false);

            tz.addEventListener('drop', function(e) {
                // 阻止默认事件 才会触发drop事件
                e.preventDefault();
                // 上传文件的信息
                console.log(e.dataTransfer.files);
            }, false);
    ```
