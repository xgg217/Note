# 函数

## 函数

  - 在函数名之后写上 `<泛型名称>`

    ```javascript
    // 通常使用 T 来代指泛型
    function name<T>(str:T):T {
      return str
    }

    name<string>();
    ```

  - 示例

    ```ts
    // 定义
    function take<T>(arr:T[], n:number):T[] {
      if(n > arr.length) {
        return arr
      }
      return arr.slice(0, n);
    }

    const arr:number[] = [4,5,45,5];
    // 调用函数
    take<number>(arr, 10)
    console.log(take<number>([1], 10));
    ```

## 泛型默认值

  - 泛型默认值 `<泛型名称 = 类型>`

    ```ts
    // 定义
    function take<T = number>(arr:T[], n:number):T[] {
      if(n > arr.length) {
        return arr
      }
      return arr.slice(0, n);
    }
    ```
