## 模块化 -- 1

  - 样式

    ```javascript
    document.addEventListener('DOMContentLoaded',function(){
      init();
    });

    const init = function() {
      // 执行
      initCompute();
    }

    const initCompute = (function () {
      "use strict";
      const oDiv = document.querySelector('div');
      let time = null;
      // 缓存数据
      const cache = {};

      const init = function() {
        // 页面初始化时 获取数据
        getAjax();
        // 执行事件
        bindEvent();
      };

      // 事件
      const bindEvent = function() {
        oDiv.addEventListener('click', event1, false);
      };

      // 执行函数
      const event1 = function(e) {
        const event = e || window.event,
          tar = event.target || event.scrElement;

        // 当缓存池 中存在数据，直接去缓存中获取数据
        if(cache[page]) {
          // 从缓存中获取数据
          getCache();
        } else {
          // 数据请求
          getAjax();
        }
      };

      // ajax
      const getAjax = function() {
        $.ajax({
          url,
          data,
          success(data) {
            // 将请求的数据进行缓存
            cache[page] = data;

            // 先显示 正在加载中
            loading.style.display = 'block';
            // 300毫秒 后显示内容
            time = setTimeout(() => {
              render(data);
              // 关闭加载中
              loading.style.display = 'none';
            }, 300)
          },
          error() {
            console.log('错误')
          }
        })
      };

      // 判断缓存中是否存在数据
      const getCache = function() {
        const data = cache[page];
        render(data)
      };


      // 渲染 HTML
      const render = function(data) {

      };

      return function() {
        init();
      }
    }());
    ```

    document.addEventListener('DOMContentLoaded',function(){
      init();
    });

    const init = function() {
      // 执行
      initCompute();
      initFun();
    }

    const initCompute = (function () {
      "use strict";
      const a = 1,
        b = 2;
      const but = document.querySelector('button');

        console.log(a + b)
        return a + b;
      };

      const minus = function (a, b) {
        console.log(21111)
        return a - b;
      };

      // 点击事件 不调用 initCompute() 也会执行
      but.addEventListener('click', function() {
        minus();
      }, false);

      }
    }());
