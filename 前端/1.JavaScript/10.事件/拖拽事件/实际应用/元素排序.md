# 元素排序

## 示例

  - 代码

    ```css
    *{
          margin: 0;
          padding: 0;
        }
        p{
          margin-top: 100px;
          margin-left: 100px;
        }
        ul{
          margin-left: 100px;
        }
        li{
          list-style: none;
          width: 200px;
          height: 30px;
          background-color: aqua;
          color: red;
          margin-bottom: 10px;
          text-align: center;
          line-height: 30px;
          cursor: pointer;
          user-select: none;
        }
        div{
          cursor: pointer;
          user-select: none;
          margin-left: 100px;
          margin-top: 80px;
          width: 300px;
          height: 300px;
          border: 5px solid #000;
        }
    ```

    ```html
    p>对元素排序</p>
      <ul>
        <li class="ys1" draggable="true">元素1</li>
        <li class="ys2" draggable="true">元素2</li>
        <li class="ys3" draggable="true">元素3</li>
        <li class="ys4" draggable="true">元素4</li>
        <li class="ys5" draggable="true">元素5</li>
      </ul>

      <div>
        <p>将元素拖进来删除</p>
      </div>
    ```

    ```js
    const ul = document.querySelector('ul'),
          li = ul.querySelectorAll('li'),
          div = document.querySelector('div');

        const bx = function(dom) {
          const len = li.length;
          for(let i = 0; i < len; i++) {
            if(dom === li[i]) {
              return li[i];
            }
          }
        };

        // 排序
        ul.addEventListener('dragstart', function(e) {
          const target = e.target;
          const index = bx(target);
          e.dataTransfer.setData("class", index.className);
          console.log(target.className);
        }, false);

        ul.addEventListener('dragover', function(e) {
        // 阻止默认事件 才会触发drop事件
        e.preventDefault();
        }, false);

        ul.addEventListener('drop', function(e) {
          const target = e.target;
          const clas = e.dataTransfer.getData("class"),
            value = this.querySelector("." + clas);

          console.log(value)
          // target.insertAdjacentHTML('beforebegin', value);
          this.insertBefore(value, target);
        }, false);



        // 删除
        div.addEventListener('dragover', function(e) {
        // 阻止默认事件 才会触发drop事件
        e.preventDefault();
        }, false);

        div.addEventListener('drop', function(e) {
        // 阻止默认事件 才会触发drop事件
          const clas = e.dataTransfer.getData("class"),
            value = ul.querySelector("." + clas);
          console.log(value);
          value.remove();
        }, false);
    ```
