# 多重回调

## promise 示例-- 多重回调

+ ES5

    ```js
    // 回掉函数
    function A() {
      function B() {
        function C() {
          function D() {}
        }
      }
    }
    ```

+ ES6

    ```js
    // 跟据上一个的状态，掉下一个函数
    A
      S1 = 'pending' ==> 'resove' new Promise()
    B
      S2 = 'pending' ==> 'resove' new Promise()
    C
      S3 = 'pending' ==> 'resove' new Promise()
    D
      S4 = 'pending' ==> 'resove' new Promise()
    ```

    ```js
    var S1 = new Promise(function (resolve, reject) {
        console.log('A开始');
        setTimeout(function () {
          resolve({value: 'A'})
        }, 2000);
      });

    S1.then((data)=>{
      console.log('A 完成了');
      return new Promise(function (resolve, reject) {
        console.log(data);
        console.log('B开始');
        setTimeout(function () {
          resolve({value: 'B'})
        }, 2000);
      });
    }, (err)=> {
      console.log(err);
    })
    .then((data) => {
      console.log('B 完成了');
      return new Promise(function (resolve, reject) {
        console.log(data);
        console.log('C开始');
        setTimeout(function () {
          resolve({value: 'C'})
        }, 2000);
      });
    }, (err) => {
      console.log(err);
    })
    .then((data) => {
      console.log('C 完成了');
      return new Promise(function (resolve, reject) {
        console.log(data);
        console.log('D开始');
        setTimeout(function () {
          resolve({value: 'D'})
        }, 2000);
      });
    }, (err) => {
      console.log(err);
    })
    .then((data) => {
      console.log('D 完成了');
    }, (err) => {
      console.log(err);
    });
    ```
