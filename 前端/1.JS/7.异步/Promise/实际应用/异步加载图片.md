# 异步加载图片

## 利用 Promise，异步加载图片

  - 代码

    ```javascript
    const imgFun = function(url) {
      return new Promise( (res, rej) => {
        const img = new Image();
        console.log('开始了');
        img.onload = function() {
          // 异步
          console.log('加载中');
          res(img);
        };
        img.onerror = function() {
          rej('加载错了渣渣');
        };
        img.src = url;
        console.log('加载咯')
      })
    };
    imgFun('http://a.hiphotos.baidu.com/image/pic/item/d009b3de9c82d158ec9917f38d0a19d8bc3e425c.jpg')
      .then(res => {
          console.log('加载好了');
          document.body.append(res)
        })
      .catch(err => {
          console.log(err);
      });

    // 执行顺序
    // 开始了
    // 加载咯
    // 加载中
    // 加载好了
    ```
