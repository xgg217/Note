# 模板方式--基于继承的设计模式

## 定义

+ 基于继承的设计模式

## 组成

1. 抽象父类
2. 具体的实现子类

## 使用

+ 通常在抽象父类中封装了子类的算法框架
+ 包括实现的一些公共方法及封装子类中所有方法的执行顺序
+ 子类通过继承这个抽象类，也继承了整个算法结构，并且可以选择重写父类

## 示例

+ 父类

  ```js
  class Beverage {
    boilWater() {
      console.log('把水煮开');
    }

    // 空方法 应该由子类重新写
    brew() {
      throw new Error('子类必须重新 brew 方法');
    }

    // 空方法 应该由子类重新写
    pourInCup() {
      throw new Error('子类必须重新 pourInCup 方法');
    }

    // 空方法 应该由子类重新写
    addCondiments() {
      throw new Error('子类必须重新 addCondiments 方法');
    }

    // 模板方法
    // 封装了子类的算法框架，指导子类的执行顺序
    init() {
      this.boilWater();
      this.brew();
      this.pourInCup();
      this.addCondiments();
    }
  }
  ```

+ 子类

  ```js
  class Coffee extends Beverage{
    constructor(){
      super();
    }

    brew() {
      console.log('用开水冲泡咖啡');
    }

    pourInCup() {
      console.log('把咖啡倒进杯子');
    }

    addCondiments() {
      console.log('加糖和牛奶');
    }
  }

  const coffee = new Coffee();
  coffee.init();
  ```

## 钩子方法

+ 隔离变化的一种常见手段
+ 在父类中容易变化的地方放置钩子，钩子可以有一个默认的实现

  ```js
  // 父类
  class Beverage {
    boilWater() {
      console.log('把水煮开');
    }

    // 空方法 应该由子类重新写
    brew() {
      throw new Error('子类必须重新 brew 方法');
    }

    // 空方法 应该由子类重新写
    pourInCup() {
      throw new Error('子类必须重新 pourInCup 方法');
    }

    // 空方法 应该由子类重新写
    addCondiments() {
      throw new Error('子类必须重新 addCondiments 方法');
    }

    // 钩子
    customerWantsCondiments() {
      return true; // 默认需要
    }

    // 模板方法
    // 封装了子类的算法框架，指导子类的执行顺序
    init() {
      this.boilWater();
      this.brew();
      this.pourInCup();
      if(this.customerWantsCondiments()) {
        // 如果返回true， 则需要
        this.addCondiments();
      }
    }
  }
  ```

  ```js
  // 子类
  class Coffee extends Beverage{
    constructor(){
      super();
    }

    brew() {
      console.log('用开水冲泡咖啡');
    }

    pourInCup() {
      console.log('把咖啡倒进杯子');
    }

    addCondiments() {
      console.log('加糖和牛奶');
    }

    customerWantsCondiments() {
      return window.confirm('需要吗？');
    }
  }

  const coffee = new Coffee();
  coffee.init();
  ```

## JS版本 模板方式

+ 父类

  ```js
  const Beverage = function(param) {
    const boilWater = () => {
      console.log('把水煮开');
    };

    const brew = param.brew || function() {
      throw new Error('必须传递 brew 方法');
    };

    const pourInCup = param.pourInCup || function() {
      throw new Error('必须传递 pourInCup 方法');
    };

    const addCondiments = param.addCondiments || function() {
      throw new Error('必须传递 addCondiments 方法');
    };

    class F{
      init() {
        boilWater();
        brew();
        pourInCup();
        addCondiments();
      }
    }
    return F;
  };
  ```

+ 子类1

  ```js
  const Coffee = Beverage({
    brew() {
      console.log('用开水冲泡咖啡');
    },
    pourInCup() {
      console.log('把咖啡倒进杯子');
    },
    addCondiments() {
      console.log('加糖和牛奶');
    }
  });
  const coffee = new Coffee();
  coffee.init();
  ```
