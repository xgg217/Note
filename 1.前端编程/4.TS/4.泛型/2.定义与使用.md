# 定义与使用

## 定义

+ 原始类型

  ```js
  type Identity<T> = T

  Identity<string>
  // string

  Identity<number>
  // number
  ```

  ```js
  type Nullable<T> = T | null

  Nullable<string>
  // string | null

  Nullable<number>
  // number | null
  ```

+ 数组

  ```js
  type Pair<T> = [T, T]

  Pair<string>
  // [string, string]

  Pair<boolean>
  // [boolean, boolean]
  ```

+ 对象

  ```js
  type Box<T> = { value:T }

  Box<number>
  // {value: number}
  ```

  ```js
  type KeyValue<K, V> = { key: K; value: V }

  KeyValue<string, number>
  // { key: string; value: number }

  KeyValue<number, boolean>
  // { key: number; value: boolean }
  ```

## 与其他类型配合使用

+ 接口：直接在名称后 `<泛型名称>`

  ```js
  interface Cb<T> {
    (n:T, i:number):boolean
  }

  function filter<T>(arr:T[], cb:Cb<T>):T[] {
    const newArr:T[] = []
    arr.forEach((n, i:number) => {
      if(cb(n, i)) {
        newArr.push(n)
      }
    });
    return newArr
  }

  const arr = filter<number>([54,6,5,46], (item) => {
    return (item > 10);
  })
  console.log(arr) // [ 54, 46 ]
  ```

+ 类型别名：直接在名称后 `<泛型名称>`

  ```js
  type Cb<T> = (n:T, i:number)=>boolean

  function filter<T>(arr:T[], cb:Cb<T>):T[] {

      }
    });
    return newArr
  }

  const arr = filter<number>([54,6,5,46], (item) => {
    return (item > 10);
  })
  console.log(arr) // [ 54, 46 ]
  ```
