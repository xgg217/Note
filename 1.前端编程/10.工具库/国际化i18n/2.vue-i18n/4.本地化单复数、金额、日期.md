# 本地化单复数、金额、日期

## 单复数

+ 使用

  ```js
  const messages = {
    en: {
      car: 'car | cars',
      apple: 'no apples | one apple | {count} apples'
    }
  }
  ```

  ```html
  <!-- 使用 -->
  <p>{{ $t('car', 1) }}</p>
  <p>{{ $t('car', 2) }}</p>

  <p>{{ $t('apple', 0) }}</p>
  <p>{{ $t('apple', 1) }}</p>
  <p>{{ $t('apple', { count: 10 }) }}</p>

  <!-- 显示效果 -->
  <p>car</p>
  <p>cars</p>

  <p>no apples</p>
  <p>one apple</p>
  <p>10 apples</p>
  ```

## 金额 / 数字 $n

+ 步骤1：国际化

  ```js
  // zh-CN.json
  {
    "message": {
      "amount": "金额 {amount}"
    }
  }

  // en-US.json
  {
    "message": {
      "amount": "amount {amount}"
    }
  }
  ```

+ 步骤2：定义规则

  ```js
  // number.ts
  export default {
    "zh-CN": {
      // 金额格式
      currency: {
        style: "currency",
        currency: "CNY",
      },
    },
    "en-US": {
      // 金额格式
      currency: {
        style: "currency",
        currency: "USD",
        // notation: "standard",
      },
    },
  } as Record<string, any>;
  ```

+ 步骤3：挂载

  ```js
  import numberFormats from "./rules/number";

  export default createI18n({
    // locale: "zh-CN", // 默认语言
    // locale: import.meta.env.VITE_DEFAULT_LOCALE, // 默认语言
    // fallbackLocale: "en-US", // 备份语言
    // legacy: false, // false：关闭兼容模式，可以使用 组合式API
    // messages: {
    //   "zh-CN": zhCN,
    //   "en-US": enUS,
    // },
    numberFormats,
    globalInjection: true,
  });
  ```

+ 步骤4：使用 `$n`

  ```html
  <div>
    {{
      t("message.amount", {
        amount: $n(1234567.89, "currency"),
      })
    }}
  </div>
  ```

  ```js
  // 中文显示效果
  金额 ¥1,234,567.89

  // 英文显示效果
  amount $1,234,567.89
  ```

## 日期时间 $d

+ 定义

  ```js
  const i18n = createI18n({
    datetimeFormats: {
      'en-US': {
        short: {
          year: 'numeric', month: 'short', day: 'numeric'
        },
        long: {
          year: 'numeric', month: 'short', day: 'numeric',
          weekday: 'short', hour: 'numeric', minute: 'numeric'
        }
      },
      'ja-JP': {
        short: {
          year: 'numeric', month: 'short', day: 'numeric'
        },
        long: {
          year: 'numeric', month: 'short', day: 'numeric',
          weekday: 'short', hour: 'numeric', minute: 'numeric', hour12: true
        }
      }
    }
  })
  ```

+ 使用

  ```html
  <p>{{ $d(new Date(), 'short') }}</p>
  <p>{{ $d(new Date(), 'long', 'ja-JP') }}</p>
  ```

+ 结果如下：

  ```html
  <p>Apr 19, 2017</p>
  <p>2017年4月19日(水) 午前2:19</p>
  ```
