# 默认语言设置

## 默认语言设置

+ 默认语言

  ```js
  // 第1 从 localStorage 中获取当前的语言
  // 第2 获取用户浏览器默认语言 navigator.language  import.meta.env.VITE_DEFAULT_LOCALE
  // 第3 环境变量中获取默认语言

  let locale = "";

  {
    // 第1 从 localStorage 中获取当前的语言
    locale = localStorage.getItem(USER_LOCALE) || "";
    if (locale) {
      return locale;
    } else if (navigator.language) {
      // 第2 获取用户浏览器默认语言 navigator.language
      locale = navigator.language;
    } else if (import.meta.env.VITE_DEFAULT_LOCALE) {
      // 第3 环境变量中获取默认语言
      locale = import.meta.env.VITE_DEFAULT_LOCALE;
    } else {
      // locale = "zh-CN";
      locale = "";
    }
  }

  export const LANGUAGE = {
    "zh-CN": "中文",
    "en-US": "English",
  } as const;

  // 获取的语言必须是支持的语言，否则默认使用 zh-CN
  if (locale in LANGUAGE) {
    return locale;
  }

  return "zh-CN";
  ```
