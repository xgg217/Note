# 时间选择器

## 限制设置

+ 规则

  + 禁止选择2024年8月1日之前的数据
  + 时间跨度为90天

  ```html
  <el-date-picker
    v-model="baTable.comSearch.form['orderInputRecord.inputTime']"
    type="datetimerange"
    range-separator="-"
    start-placeholder="Start date"
    :disabled-date="disabledDate"
    end-placeholder="End date"
    value-format="YYYY-MM-DD HH:mm:ss"
    @calendar-change="onCalendarChange"
  />

  <script lang="ts" setup>

    // 时间限制
    const { disabledDate, onCalendarChange } = (() => {
      const tempTime = 3600 * 1000 * 24;
      const tempTime90 = tempTime * 90;

      const minDate = new Date(2024, 7, 1).getTime();
      let chooseDay: number = Date.now(); // 选择的第一个日期

      // 选中的第一个时间
      const onCalendarChange = (val: [Date, Date]) => {
        chooseDay = val[0].getTime();
      };

      const disabledDate = (time: Date) => {
        const timeVal = time.getTime();

        // 禁止选择2024年8月1日之前的数据
        if (timeVal <= minDate - tempTime) {
          return true;
        }

        // 最小时间
        const nowMinDate = Math.max(chooseDay - tempTime90, minDate);
        // 最大时间
        const nowMaxDate = chooseDay + tempTime90;

        if (timeVal < nowMinDate) {
          return true;
        }

        if (timeVal > nowMaxDate) {
          return true;
        }
        return false;
      };

      return {
        disabledDate,
        onCalendarChange,
      };
    })();
  </script>
  ```

+ 限制最大时间间隔为15天

  ```html
  <el-date-picker
    v-model="ruleForm2.val"
    type="daterange"
    range-separator="-"
    :start-placeholder="$t('StartDate')"
    :end-placeholder="$t('EndDate')"
    :disabled-date="disabledDateMethod"
    @calendar-change="onCalendarChange"
  />
  ```

  ```js
  const tempTime = 3600 * 1000 * 24;
  const tempTime15 = tempTime * 15;
  let chooseDay: number = Date.now(); // 选择的第一个日期

  // 选中的第一个时间
  const onCalendarChange = (val: [Date, Date]) => {
    chooseDay = val[0].getTime();
  };

  const disabledDateMethod = (time: Date) => {
    const timeVal = time.getTime();

    // 最大时间
    const nowMaxDate = chooseDay + tempTime15;
    const nowMinDate = chooseDay - tempTime15;

    // 时间范围
    if (timeVal >= nowMaxDate) {
      return true;
    }

    if (timeVal <= nowMinDate) {
      return true;
    }

    return false;
  };
  ```
