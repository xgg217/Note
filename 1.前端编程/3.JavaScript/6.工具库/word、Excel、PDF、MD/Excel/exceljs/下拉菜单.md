# 下拉菜单

## 概述

+ formulae 数组设置

  ```js
  const enumMap = { 1: "aa", 2: "bb" };
  const values = Object.values(enumMap); // ["aa", "bb"]
  const formula = `"${values.join(',')}"`; // "aa,bb"
  ```

## 方式1

+ 下拉菜单
+ 格式 formulae: [`"'下拉菜单1,下拉菜单1'"`]

  ```js
  const wb = new ExcelJS.Workbook();
  const Sheet1 = wb.addWorksheet("Sheet1");

  Sheet1.columns = [
    { header: "ID", key: "id", width: 20 },
    { header: "姓名", key: "name", width: 30 },
  ];

  const Sheet1_data = [
    { id: 1, name: "II" },
    { id: 2, name: "ICMS" },
  ];
  Sheet1.addRows(Sheet1_data);

  Sheet1.dataValidations.add("B2:B100", {
    type: "list", // 下拉列表类型
    allowBlank: true, // 允许空值
    showErrorMessage: true,
    errorTitle: "输入错误",
    error: "只能选择男/女",
    formulae: ['"II,ICMS,AA"'], // 选项：英文逗号分隔，外层双引号包裹
  });

  wb.xlsx.writeBuffer().then(data => {
    const blob = new Blob([data], {
      type: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
    });
    saveAs(blob, "税费导出模板.xlsx");
  });
  ```



## 方式2 引用其他表数据

+ 引用其他表数据

  ```js
  const wb = new ExcelJS.Workbook();
  const Sheet1 = wb.addWorksheet("Sheet1");


  Sheet1.columns = [
    { header: "ID", key: "id", width: 20 },
    { header: "姓名", key: "name", width: 30 },
  ];

  const Sheet1_data = [
    { id: 1, name: "II" },
    { id: 2, name: "ICMS" },
  ];
  Sheet1.addRows(Sheet1_data);

  // 其他表
  {
    // 新建“选项表”存放数据源
    const optionsSheet = wb.addWorksheet("选项表");
     optionsSheet.state = "hidden"; // 隐藏
    optionsSheet.getColumn("A").values = ["", "II", "ICMS", "未知"]; // A列：性别选项

    Sheet1.dataValidations.add("B2:B100", {
      type: "list", // 下拉列表类型
      allowBlank: true, // 允许空值
      showErrorMessage: true,
      errorTitle: "输入错误",
      formulae: ["=选项表!$A$2:$A$4"], // 选项：英文逗号分隔，外层双引号包裹
    });
  }

  wb.xlsx.writeBuffer().then(data => {
    const blob = new Blob([data], {
      type: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
    });
    saveAs(blob, "税费导出模板.xlsx");
  });
  ```
