# 实际使用

## 表格导入时间格式化

+ 表格导入时间格式化

  ```js
  /**
   * @description excel表格导入时间格式化
   * @param numb
   * @param format1
   * @param format2
   */
  export const formatDateExcel = (
    numb: number,
    format1: string,
    format2: string,
  ): string | boolean => {
    const old = numb - 1;
    if (isNaN(old)) {
      return true;
    }
    const t = Math.round((old - Math.floor(old)) * 24 * 60 * 60);
    const time = new Date(1900, 0, old, 0, 0, t);
    const year = time.getFullYear();
    const month = time.getMonth() + 1;
    const date = time.getDate();
    const HH = time.getHours();
    const MM = time.getMinutes();
    const ss = time.getSeconds();
    return `${year}${format1}${month.toString().padStart(2, "0")}${format1}${date.toString().padStart(2, "0")} ${HH.toString().padStart(2, "0")}${format2}${MM.toString().padStart(2, "0")}${format2}${ss.toString().padStart(2, "0")}`;
  };

  ```
