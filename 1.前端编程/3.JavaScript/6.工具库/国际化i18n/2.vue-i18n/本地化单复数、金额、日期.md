# 本地化单复数、金额、日期

## 单复数

+ 使用

  ```js
  const messages = {
    en: {
      car: 'car | cars',
      apple: 'no apples | one apple | {count} apples'
    }
  }
  ```

  ```html
  <!-- 使用 -->
  <p>{{ $t('car', 1) }}</p>
  <p>{{ $t('car', 2) }}</p>

  <p>{{ $t('apple', 0) }}</p>
  <p>{{ $t('apple', 1) }}</p>
  <p>{{ $t('apple', { count: 10 }) }}</p>

  <!-- 显示效果 -->
  <p>car</p>
  <p>cars</p>

  <p>no apples</p>
  <p>one apple</p>
  <p>10 apples</p>
  ```

## 时间

+ 定义

  ```js
  const i18n = createI18n({
    datetimeFormats: {
      'en-US': {
        short: {
          year: 'numeric', month: 'short', day: 'numeric'
        },
        long: {
          year: 'numeric', month: 'short', day: 'numeric',
          weekday: 'short', hour: 'numeric', minute: 'numeric'
        }
      },
      'ja-JP': {
        short: {
          year: 'numeric', month: 'short', day: 'numeric'
        },
        long: {
          year: 'numeric', month: 'short', day: 'numeric',
          weekday: 'short', hour: 'numeric', minute: 'numeric', hour12: true
        }
      }
    }
  })
  ```

+ 使用

  ```html
  <p>{{ $d(new Date(), 'short') }}</p>
  <p>{{ $d(new Date(), 'long', 'ja-JP') }}</p>
  ```

+ 结果如下：

  ```html
  <p>Apr 19, 2017</p>
  <p>2017年4月19日(水) 午前2:19</p>
  ```

## 数字格式化

+ 定义

  ```js
  const i18n = createI18n({
    numberFormats: {
      'en-US': {
        currency: {
          style: 'currency', currency: 'USD', notation: 'standard'
        },
        decimal: {
          style: 'decimal', minimumFractionDigits: 2, maximumFractionDigits: 2
        },
        percent: {
          style: 'percent', useGrouping: false
        }
      },
      'ja-JP': {
        currency: {
          style: 'currency', currency: 'JPY', useGrouping: true, currencyDisplay: 'symbol'
        },
        decimal: {
          style: 'decimal', minimumSignificantDigits: 3, maximumSignificantDigits: 5
        },
        percent: {
          style: 'percent', useGrouping: false
        }
      }
    }
  })
  ```

+ 使用

  ```html
  <p>{{ $n(10000, 'currency') }}</p>
  <p>{{ $n(10000, 'currency', 'ja-JP') }}</p>
  <p>{{ $n(10000, 'currency', 'ja-JP', { useGrouping: false }) }}</p>
  <p>{{ $n(987654321, 'currency', { notation: 'compact' }) }}</p>
  <p>{{ $n(0.99123, 'percent') }}</p>
  <p>{{ $n(0.99123, 'percent', { minimumFractionDigits: 2 }) }}</p>
  <p>{{ $n(12.11612345, 'decimal') }}</p>
  <p>{{ $n(12145281111, 'decimal', 'ja-JP') }}</p>
  ```

+ 结果如下

  ```html
  <p>$10,000.00</p>
  <p>￥10,000</p>
  <p>￥10000</p>
  <p>$988M</p>
  <p>99%</p>
  <p>99.12%</p>
  <p>12.12</p>
  <p>12,145,000,000</p>
  ```
