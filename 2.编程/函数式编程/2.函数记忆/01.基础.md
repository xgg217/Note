# 函数记忆

## 形式

+ 代码

  ```js
  function memorize(fun) {
    const cache = {};
    return function (...ags) {
      const key = ags.length + ags.join("");
      if(cache[key]) {
        return cache[key];
      }
      cache[key] = fun.apply(this, ags);
      return cache[key];
    }
  }
  ```


## 示例

+ 斐波那契数列

  ```js
  // 未使用函数记忆
  function factorial(n) {
    if((n === 0) || (n == 1)) {
      return 1;
    }
    return n + factorial(n - 1)
  }

  factorial(500)
  ```
