# 分段读取文件

## 示例

*   代码

    ```css
    img{
                display: inline-block;
                width: 100px;
                height: 100px;
                border: 1px solid #000;
            }
            input{
                /* opacity: 0;
                display: none; */
            }
            div{
                display: none;
                margin-top: 20px;
                width: 200px;
                height: 20px;
                background-color:aquamarine
            }
            div p{
                width: 0%;
                height: 20px;
                background-color: red;
            }
            p span{
                border: 1px solid #000;
                cursor: pointer;
                display: none;
            }
    ```

    ```html
    <label for="fil">
            <img src="" alt="">
        </label>
        <input type="file" id="fil">
        <div class="ddd"><p></p></div>
        <p><span>暂停</span></p>
    ```

    ```javascript
    const oInput = document.querySelector('input'),
            oImg = document.querySelector('img'),
            oLabel = document.querySelector('label'),
            oDiv = document.querySelector('.ddd'),
            oPe = oDiv.querySelector('p'),
            span = document.querySelector('span');

        
        oInput.addEventListener('change', function(){
            console.log(this.files);
            // 分段上传
            const fileNumber = 1024 * 1024; // 每次上传 1M 大小
            let filstFile = 0; // 每次读取的 开始部分
                // allFile = '';
            const file = this.files[0],
                allFile = Math.ceil(file.size / fileNumber), // 一共多少片
                fileSize = file.size; // 总长度


            const reader = new FileReader();
            if(file) {
                // 触发 load 事件
                // reader.readAsDataURL(file);
                reader.readAsArrayBuffer(file);
            } else {
                return;
            }

            const jxdq = function(start) {
                if(!file.slice) {
                    // 如果不存在这个方法
                    return;
                }
                let end = '';
                const startM = start * fileNumber; // 本次开始的位置
                if((startM + fileNumber) < fileSize ) {
                    // 继续读取
                    end = startM + fileNumber;
                } else {
                    // 最后一段
                    end = fileSize;
                }
                
                reader.readAsArrayBuffer(file.slice(startM, end));

            };
            oDiv.style.display = 'block';
            span.style.display = 'block';

            // 开始读取
            reader.addEventListener('loadstart', function(){
                console.log('读取开始');
               
            }, false);

            // 读取被中断
            reader.addEventListener('abort', function(){
                console.log('取被中断');
            }, false);

            // 读取完成
            reader.addEventListener("load", function (e) {
                
                console.log(e);
                if(allFile > filstFile) {
                    // 未传输完成, 继续上传
                    console.log(filstFile)
                    jxdq(filstFile);
                    
                    const sl = Math.ceil((filstFile / allFile) * 100);
                    oPe.style.width = sl + '%';
                    oPe.textContent =  sl + '%';
                    filstFile++;
                    console.log('继续上传');
                } else {
                    console.log('读取完成');
                    oPe.style.width = '100%';
                    oPe.textContent = '100%';
                }
            }, false);
        }, false);
    ```
