# 函数

## 对象函数

*   对象函数

    ```javascript
    interface User{
      name:string
      age:number

      // 定义方式1
      sayHello:()=>void // 该函数没有参数，函数返回类型是 void

      // 定义方式2:
      getName():void
    }

    let user:User = {
      name: '小刚刚',
      age: 11,
      sayHello() {
        console.log('侬好')
      },
      getName() {
        console.log('小刚刚')
      }
    }
    ```

## 函数

*   函数

    ```javascript
    // Cb 就相当于函数名
    interface Cb{
      // 该函数接受一个参数，返回值为 布尔值
      (n:number):boolean
    }

    function setType(arr:number[], func:Cb) {
      let s = 0;
      arr.forEach(item => {
        if(func(item)) {
          s += item;
        }
      })
      return s;
    }

    console.log(setType([57,8,80], (item) => {
      return (item % 2) !== 0
    }));
    ```
