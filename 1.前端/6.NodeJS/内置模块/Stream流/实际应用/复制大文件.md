# 复制大文件

## 方法1

  - 方法1：

      - 使用 `fs.readFile 和 fs.writeFile`

      - 缺点：内存吃不消

## 方法2

  - 方法2 使用流：即读多少 写入多少

    ```javascript
    const path = require("path");
    const fs = require("fs");

    const filend = path.resolve(__dirname, "./files/abc.txt");
    const filend1 = path.resolve(__dirname, "./files/abc1.txt");

    // 读文件
    const read = fs.createReadStream(filend);

    // 写
    const ws = fs.createWriteStream(filend1);

    read.on("data", (chunk) => {
      console.log("读取了一部分文件");
      const flag = ws.write(chunk);
      if(!flag) {
        // 表示下一次写入会找出背压
        read.pause(); // 暂停读取
      }
    });


    read.on("end", () => {
      console.log("读取结束");
      ws.end();
    })

    ws.on("drain", () => {
      console.log("可以写入");
      read.resume();
    })
    ws.on("close", () => {
      console.log("结束了");
    })
    ```

## 方式3

  - 方式3：方式2的简写，即 使用 管道符

    ```javascript
    const path = require("path");
    const fs = require("fs");

    const filend = path.resolve(__dirname, "./files/abc.txt");
    const filend1 = path.resolve(__dirname, "./files/abc1.txt");

    // 读文件
    const read = fs.createReadStream(filend, {
      // encoding: "utf-8",
    });

    // 写
    const ws = fs.createWriteStream(filend1);

    read.pipe(ws);

    read.on("close", () => {
      console.log("读取结束");
      ws.end();
    })
    ```
