# 父到子

## 父--> 子 传值

  - 静态

    ```javascript
    <scoring1  name="commodityClick"></scoring1>
    ```

  - 动态传值

    ```javascript
    // 父组件
    <scoring1 :aa.sync="isAdditional" ></scoring1>

    data = {
      isAdditional: true
    }
    ```

    ```javascript
    // 子组件
    props = {
      aa: {
        type: Boolean,
        default: false,
      }
    }
    ```

  - 监听数据变化

    ```javascript
    props = {
      contetValue: {
        type: String,
        default: '0',
        handler:function(){
          console.log('hello world')
        }
      },
    }
    data = {
      list: []
    }

    computed = {
      list() {
        this.contetValue;
        this.list;
      }
    }
    ```
